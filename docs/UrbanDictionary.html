def define(word)
  params = {
    term: word
  }
  @client = HTTPClient.new if @client.nil?
  response = JSON.parse(@client.get(URI.parse(URL), params).body)
  ret = []
  response['list'].each do |hash|
    ret << Slang.new(hash)
  end
  ret
end
